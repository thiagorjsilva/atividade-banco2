    Nacionalidade VARCHAR(50)
);

CREATE TABLE Livros (
    ID INT PRIMARY KEY,
    Titulo VARCHAR(100),
    AnoPublicacao YEAR,
    Preco DECIMAL(5,2),
    ID_Autor INT
